{
   "Mercury",
   "Venus",
   "Earth",
   "Mars",
   "Jupiter",
   "Saturn",
   "Uranus",
   "Neptune"
};

planets.Sort();

foreach(string planet in planets)
{
   Console.WriteLine(planet);
}
